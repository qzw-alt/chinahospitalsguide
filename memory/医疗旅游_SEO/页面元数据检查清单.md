## Schema标记建议

### 需要添加的Schema类型

```json
// Organization Schema (添加到首页)
{
  "@context": "https://schema.org",
  "@type": "Organization",
  "name": "China Hospitals Guide",
  "url": "https://chinahospitalsguide.com",
  "logo": "https://chinahospitalsguide.com/logo.png",
  "contactPoint": {
    "@type": "ContactPoint",
    "contactType": "customer service",
    "availableLanguage": ["English", "Chinese"]
  }
}

// MedicalWebPage Schema (添加到治疗页)
{
  "@context": "https://schema.org",
  "@type": "MedicalWebPage",
  "name": "Cancer Treatment in China",
  "medicalAudience": {
    "@type": "MedicalAudience",
    "audienceType": "Patient"
  }
}

// FAQPage Schema (添加到FAQ页)
{
  "@context": "https://schema.org",
  "@type": "FAQPage",
  "mainEntity": [{
    "@type": "Question",
    "name": "Is it safe to have surgery in China?",
    "acceptedAnswer": {
      "@type": "Answer",
      "text": "..."
    }
  }]
}
```
